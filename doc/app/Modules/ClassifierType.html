def self.all_types
  unwind = {"$unwind"=>"$code"}
  group_stage = {
      "$group" => {
          "_id" => nil,
          "codes" => { "$push" => "$code" }
      }
  }
  project={
      "$project"=>{"_id"=>0, "code"=>"$codes"}
  }
  self.collection.aggregate([unwind, group_stage, project])[0][:code]
end
