def self.get_hash_links_by_town(town)
  links = self.get_links_by_town(town)
  res = []
  links.each do |link|
    # TODO: maybe change links query
    # today category_title not show on site

    tmp_hash = {
      # category_title: cat_title,
      href: link.value,
      title: link.title,
      description: link.description
    }
    res << tmp_hash
  end
  res
end
